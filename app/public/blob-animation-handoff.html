Flutter Implementation
Dart blob_background.dart 
// Drop-in widget. Accepts active: bool.
// Wrap in a Stack, positioned behind the dark glass surface.

import 'dart:ui';
import 'package:flutter/material.dart';

class BlobBackground extends StatefulWidget {
  final bool active;
  const BlobBackground({super.key, this.active = true});
  @override
  State<BlobBackground> createState() => _BlobBackgroundState();
}

class _BlobBackgroundState extends State<BlobBackground>
    with TickerProviderStateMixin {

  late AnimationController _c1, _c2, _c3;
  late Animation<double> _b1x, _b1y, _b2x, _b2y, _b3x, _b3y;

  double get _spd => widget.active ? 1.0 : 2.8;

  @override
  void initState() {
    super.initState();
    _c1 = _ctrl(7000)..repeat();
    _c2 = _ctrl(9000);
    _c3 = _ctrl(6000);
    Future.delayed(const Duration(seconds: 1), () { if (mounted) _c2.repeat(); });
    Future.delayed(const Duration(seconds: 2), () { if (mounted) _c3.repeat(); });
    _wire();
  }

  AnimationController _ctrl(int ms) =>
      AnimationController(vsync: this,
          duration: Duration(milliseconds: (ms * _spd).round()));

  // Build a TweenSequence from web-style keyframe arrays.
  // Each segment gets its own easeInOut — matching Framer Motion's behaviour.
  Animation<double> _kf(AnimationController c, List<double> v) =>
      TweenSequence<double>([
        for (var i = 0; i < v.length - 1; i++)
          TweenSequenceItem(
            tween: Tween<double>(begin: v[i], end: v[i+1])
                .chain(CurveTween(curve: Curves.easeInOut)),
            weight: 1,
          ),
      ]).animate(c);

  void _wire() {
    _b1x = _kf(_c1, [0,  50,   10,  70, 0]);
    _b1y = _kf(_c1, [0,  15,  -10,   5, 0]);
    _b2x = _kf(_c2, [0, -40,   -5, -60, 0]);
    _b2y = _kf(_c2, [0,  10,  -15,   5, 0]);
    _b3x = _kf(_c3, [0,  25,  -30,  15, 0]);
    _b3y = _kf(_c3, [0, -10,   20,  -5, 0]);
  }

  @override
  void didUpdateWidget(BlobBackground old) {
    super.didUpdateWidget(old);
    if (old.active != widget.active) {
      _c1.duration = Duration(milliseconds: (7000 * _spd).round());
      _c2.duration = Duration(milliseconds: (9000 * _spd).round());
      _c3.duration = Duration(milliseconds: (6000 * _spd).round());
      _wire();
    }
  }

  @override
  void dispose() { _c1.dispose(); _c2.dispose(); _c3.dispose(); super.dispose(); }

  @override
  Widget build(BuildContext ctx) {
    return AnimatedOpacity(
      opacity: widget.active ? 1.0 : 0.55,
      duration: const Duration(milliseconds: 800),
      child: ShaderMask(
        // Fade top ~10 dp transparent → opaque (hides hard clip at widget edge)
        shaderCallback: (b) => const LinearGradient(
          begin: Alignment.topCenter, end: Alignment.bottomCenter,
          colors: [Colors.transparent, Colors.black],
          stops: [0.0, 0.05],
        ).createShader(b),
        blendMode: BlendMode.dstIn,
        child: ImageFiltered(
          imageFilter: ImageFilter.blur(sigmaX: 28, sigmaY: 28),
          child: AnimatedBuilder(
            animation: Listenable.merge([_c1, _c2, _c3]),
            builder: (ctx, _) => LayoutBuilder(
              builder: (ctx, c) => Stack(
                clipBehavior: Clip.none,
                children: [
                  Positioned(left: _b1x.value,                      top: _b1y.value, child: _blob(200, const Color(0xFFFF6B5B))),
                  Positioned(right: -_b2x.value,                     top: _b2y.value, child: _blob(180, const Color(0xFF5C63D6))),
                  Positioned(left: c.maxWidth * 0.3 + _b3x.value,  top: 10 + _b3y.value, child: _blob(160, const Color(0xFF8C92FF))),
                ],
              ),
            ),
          ),
        ),
      ),
    );
  }

  Widget _blob(double s, Color c) => SizedBox(
    width: s, height: s,
    child: DecoratedBox(decoration: BoxDecoration(shape: BoxShape.circle, color: c)),
  );
}
Implementation Notes
⚡ Blur is on the blob layer, not backdrop. Use ImageFiltered(imageFilter: ImageFilter.blur(...)) wrapping all three blobs together — not BackdropFilter, which blurs what's behind the widget. This is the key to the effect.
🎨 The dark surface above needs its own opacity. The glass panel uses Color(0xD1121827) — that's #121827 at ~82% alpha (0xD1 ≈ 209/255). The remaining 18% is what lets the blobs show through. Also apply BackdropFilter(blur 4px) on this surface for the frosted glass feel.
🪄 Per-segment easing. The web uses Framer Motion's easeInOut applied to each keyframe segment individually — not the whole animation. That's why TweenSequenceItem wraps each segment with its own CurveTween(curve: Curves.easeInOut).
📐 Speed change on active toggle. When recording stops, blobs slow to 2.8× their normal cycle time. Update AnimationController.duration in didUpdateWidget and rebuild the TweenSequence — the animation continues from its current position.
🎭 Entry/exit animation. The full widget bar slides in from the bottom using a spring: stiffness 380, damping 38. In Flutter: CurvedAnimation(curve: Curves.elasticOut) with ~400ms duration, or a custom SpringSimulation.
